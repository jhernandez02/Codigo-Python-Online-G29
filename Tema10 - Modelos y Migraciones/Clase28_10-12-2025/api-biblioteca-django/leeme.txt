# Crear el entorno virtual
python -m venv .venv

# Activar el entorno virtual
.venv\Scripts\activate

# Instalar los paquetes
pip install -r requirements.txt

# Crear el proyecto en django
django-admin startproject app ../api-biblioteca-django

# Crea el archivo .env, en la raíz del proyecto, con el siguiente contenido:
PGHOST='tu-host'
PGDATABASE='bibliotecadb'
PGUSER='neondb_owner'
PGPASSWORD='tu_password'
PGPORT=5432

# Ejecutar las migraciones por defecto
python manage.py migrate

# Crear el usuario administrador (admin:123456)
python manage.py createsuperuser

# Crear la aplicación biblioteca
python manage.py startapp biblioteca
